      package main

      import (
        "fmt"
        "net/http"
        "io/ioutil"
      )

      func main() {
        apiKey := "YOUR_API_KEY"

        // Semua nama
        urlAll := "https://api2.teamhackbotsindo.my.id/api/asmaulHusna"
        client := &http.Client{}
        reqAll, _ := http.NewRequest("GET", urlAll, nil)
        reqAll.Header.Set("x-api-key", apiKey)
        respAll, _ := client.Do(reqAll)
        defer respAll.Body.Close()
        bodyAll, _ := ioutil.ReadAll(respAll.Body)
        fmt.Println("Semua Asmaul Husna:", string(bodyAll))

        // Detail per nomor
        urlDetail := "https://api2.teamhackbotsindo.my.id/api/asmaulHusna?no=5"
        reqDetail, _ := http.NewRequest("GET", urlDetail, nil)
        reqDetail.Header.Set("x-api-key", apiKey)
        respDetail, _ := client.Do(reqDetail)
        defer respDetail.Body.Close()
        bodyDetail, _ := ioutil.ReadAll(respDetail.Body)

      }
