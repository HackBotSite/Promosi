      import requests

      apiKey = "YOUR_API_KEY"
      url = "https://api2.teamhackbotsindo.my.id/api/resi?resi=JP1234567890&courier=jne"

      print(resp.json())
      

      package main

      import (
        "fmt"
        "net/http"
        "io/ioutil"
      )

      func main() {
        apiKey := "YOUR_API_KEY"
        url := "https://api2.teamhackbotsindo.my.id/api/resi?resi=JP1234567890&courier=jne"

        client := &http.Client{}
        req, _ := http.NewRequest("GET", url, nil)
        req.Header.Set("x-api-key", apiKey)

        resp, _ := client.Do(req)
        defer resp.Body.Close()

        body, _ := ioutil.ReadAll(resp.Body)
        fmt.Println(string(body))
      }
