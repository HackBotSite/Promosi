      package main

      import (
        "fmt"
        "net/http"
        "io/ioutil"
      )

      func main() {
        client := &http.Client{}
        req, _ := http.NewRequest("GET", "https://api.teamhackbotsindo.my.id/api/doa", nil)
        req.Header.Set("x-api-key", "sk-demo-api-key-1234567890abcdef")

        resp, _ := client.Do(req)
        defer resp.Body.Close()

        body, _ := ioutil.ReadAll(resp.Body)
        fmt.Println(string(body))
      }
